127 /*
128  @brief Compute the local mass matrix with a linear source term
129  @details It computes the local mass matrix with a linear source term for a
130  linear finite element on an N dimensional simplex given the measure of the
131  simplex, and the vector of source terms In practice it computes the matrix
132  given by \f[ M_{ij}=\int_{\Omega} \phi_i \phi_j + \int_{\Omega} \phi_i(x)
133  \phi_j(x) \sum_{k=0}^{N} \phi_k(x)\source_k \f] where the \f$\phi_i\f$ are the
134  basis functions of a linear finite element on the simplex of dimension N.
135  @tparam N The dimension of the space
136  @param measure The measure of the simplex
137  @param source The vector of source terms
138  @return The elemental mass matrix
139  @note I am using the formula
140  \f[
141  \int_{\Omega} \phi_i^m \phi_j^n \phi_k(x)^p = \frac{m!n!p!N!}{(m+n+p+N)!}
142  \f]
143  */
144 template <unsigned int N>
145 auto computeLocalMassWithSource(double measure,
146  const Eigen::Matrix<double, N + 1, 1> &source)
147  -> Eigen::Matrix<double, N + 1, N + 1> {
148  Eigen::Matrix<double, N + 1, N + 1> localMass;
149  localMass.setZero();
150  // I have not found a better way to do it, sorry
151  constexpr double noEqual =
152  factorial<N>() / static_cast<double>(factorial<N + 3u>());
153  constexpr double twoEqual = 2.0 * noEqual;
154  constexpr double allEqual = 6.0 * noEqual;
155  for (auto i = 0u; i < N + 1; ++i) {
156  for (auto j = 0u; j <= i; ++j) {
157  for (auto k = 0u; k < N + 1; ++k) {
158  if (i == j && j == k) {
159  localMass(i, j) += measure * allEqual * source(k);
160  } else if (i == j || j == k || i == k) {
161  localMass(i, j) += measure * twoEqual * source(k);
162  } else {
163  localMass(i, j) += measure * noEqual * source(k);
164  }
165  }
166  }
167  }
168  return localMass.template selfadjointView<Eigen::Lower>();
169 }
