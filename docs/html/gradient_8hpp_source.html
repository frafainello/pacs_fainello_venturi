17 // the function computeGradCoeff returns the inverse of the Jacobian matrix D,
18 // which is used to compute the gradients of the shape functions in the actual
19 // element coordinates. This step is essential in finite element methods to
20 // transform local (reference) gradients to global (actual) gradients.
21 template <unsigned int N>
22 auto computeGradCoeff(const Eigen::Matrix<double, N, N + 1> &vertices)
23  -> Eigen::Matrix<double, N, N> {
24  using MatrixNd = Eigen::Matrix<double, N, N>;
25  // Compute the matrix of differences between vertices
26  MatrixNd D;
27  for (auto i = 0u; i < N; ++i) {
28  D.row(i) = vertices.col(i + 1) - vertices.col(0); // D=B_trasposto
29  }
30  // Compute the inverse of the matrix of differences
31  return D.inverse();
32 }

54  const Eigen::Matrix<double, N, N> &gradCoeff,
55  double const &measure,
56  double const &factor = 1.0)
57  -> Eigen::Matrix<double, N + 1, N + 1> {
58  Eigen::Matrix<double, N, N + 1> gradCoeffExt;
59  Eigen::Matrix<double, N + 1, N + 1> localStiffness;
60 
61  // Constructs gradCoeffExt, an extended gradient coefficient matrix of size N x (N + 1):
62  // The first column is the negative sum of the rows of gradCoeff.
63  // The remaining columns are copied directly from gradCoeff.
64  gradCoeffExt.template rightCols<N>() = gradCoeff;
65  gradCoeffExt.template leftCols<1>() = -gradCoeff.rowwise().sum();
66 
67  // compute the local stiffness matrix
68  for (auto i = 0u; i < N + 1; ++i) {
69  localStiffness(i, i) =
70  factor * measure * gradCoeffExt.col(i).dot(gradCoeffExt.col(i));
71  for (auto j = 0u; j < i; ++j) { // compute lower triangular part
72  auto value =
73  factor * measure * gradCoeffExt.col(i).dot(gradCoeffExt.col(j));
74  localStiffness(i, j) = value;
75  localStiffness(j, i) = value;
76  }
77  }
78  return localStiffness;
79 }

175 /*
176  @brief Compute the local mass matrix with a linear source term
177  @details It computes the local mass matrix with a linear source term for a
178  linear finite element on an N dimensional simplex given the measure of the
179  simplex, and the vector of source terms In practice it computes the matrix
180  given by \f[ M_{ij}=\int_{\Omega} \phi_i \phi_j + \int_{\Omega} \phi_i(x)
181  \phi_j(x) \sum_{k=0}^{N} \phi_k(x)\source_k \f] where the \f$\phi_i\f$ are the
182  basis functions of a linear finite element on the simplex of dimension N.
183  @tparam N The dimension of the space
184  @param measure The measure of the simplex
185  @param source The vector of source terms
186  @return The elemental mass matrix
187  @note I am using the formula
188  \f[
189  \int_{\Omega} \phi_i^m \phi_j^n \phi_k(x)^p = \frac{m!n!p!N!}{(m+n+p+N)!}
190  \f]
191  */
192 template <unsigned int N>
193 auto computeLocalMassWithSource(double measure,
194  const Eigen::Matrix<double, N + 1, 1> &source)
195  -> Eigen::Matrix<double, N + 1, N + 1> {
196  Eigen::Matrix<double, N + 1, N + 1> localMass;
197  localMass.setZero();
198  // I have not found a better way to do it, sorry
199  constexpr double noEqual =
200  factorial<N>() / static_cast<double>(factorial<N + 3u>()); // 1! = 1
201  constexpr double twoEqual = 2.0 * noEqual; // 2! = 2
202  constexpr double allEqual = 6.0 * noEqual; // 3! = 6 (threeEqual)
203  for (auto i = 0u; i < N + 1; ++i) {
204  for (auto j = 0u; j <= i; ++j) {
205  for (auto k = 0u; k < N + 1; ++k) {
206  if (i == j && j == k) {
207  localMass(i, j) += measure * allEqual * source(k);
208  } else if (i == j || j == k || i == k) {
209  localMass(i, j) += measure * twoEqual * source(k);
210  } else {
211  localMass(i, j) += measure * noEqual * source(k);
212  }
213  }
214  }
215  }
216  return localMass.template selfadjointView<Eigen::Lower>();
217 }
