38  bool updateSolution(const int rank, const int size) {
39 
40  // Reset attributes for each update
41  reset_attributes();
42 
43  // Prepare to parallelization. The last process handles the remainder
44  int n = this->mesh.getNumElements();
45  int local_n = n / size;
46  int start = rank * local_n;
47  int end = (rank == size - 1) ? n : start + local_n;
48 
49  // Variables for timing
50  double start_time, end_time, elapsed_time;
51  start_time = MPI_Wtime();
52 
53  // Compute local rhs for each element and update global rhs
54  for (int i = start; i < end; i++) {
55  computeLocalRhs(i);
56  updateGlobalRhs(i);
57  }
58 
59  // Sum global rhs values from all processes to rank 0
60  Values global_rhs(rhs.size());
61  MPI_Reduce(rhs.data(), global_rhs.data(), rhs.size(), MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);
62 
63  // Timing computation
64  end_time = MPI_Wtime();
65  elapsed_time = end_time - start_time;
66  if (rank == 0){
67  std::cout << "Time taken for rhs with " << size << " cores is " << elapsed_time << " s" << std::endl;
68  }
69 
70  // Sum global grad_w values from all processes to rank 0
71  Eigen::Matrix<double, Eigen::Dynamic, PHDIM> global_grad_w(grad_w.size() / PHDIM, PHDIM);
72  MPI_Reduce(grad_w.data(), global_grad_w.data(), grad_w.size(), MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);
73 
74  bool localConverged = true; // Initialize to true for ranks other than 0
75 
76  // rank 0 updates the solution
77  if (rank == 0) {
78  rhs = global_rhs;
79  grad_w = global_grad_w;
80 
81  // Apply boundary conditions with TGV method
82  for (int idx : mesh.getBoundaryNodes()) {
83  rhs(idx) = 0.0 * 1e40;
84  }
85 
86  // Solve the linear system for z
87  z = solver.solve(rhs);
88 
89  // Incremental update of the solution w
90  w += z;
91  std::cout << "z : " << z.norm() << std::endl;
92  updateLagrangians(z);
93 
94  // Check for convergence
95  localConverged = (z.norm() < tol);
96  }
97 
98  // Broadcast the updated value of w from rank 0 to all other ranks
99  MPI_Bcast(w.data(), w.size(), MPI_DOUBLE, 0, MPI_COMM_WORLD);
100 
101  return localConverged;
102  }

361 class LagrangianEikonal : public EikonalEquation<PHDIM> {
362 public:
363  using Traits = Eikonal::Eikonal_traits<PHDIM, INTRINSIC_DIM>;
364  using Index = typename Traits::Index;
365  using Indexes = typename Traits::Indexes;
366  using Nodes = typename Traits::Nodes;
367  using Values = typename Traits::Values;
368 
369  // Constructor for Lagrangian Eikonal
370  LagrangianEikonal(MeshData<PHDIM>& mesh,
371  Values& w,
372  const Eigen::SparseMatrix<double>& stiffnessMatrix,
373  const Eigen::SparseLU<Eigen::SparseMatrix<double>>& solver,
374  double r,
375  Eigen::Matrix<double, Eigen::Dynamic, PHDIM>& lagrangians
376  )
377  : EikonalEquation<PHDIM>(mesh, w, stiffnessMatrix, solver), r(r), lagrangians(lagrangians) {}
378 
379  // Compute the stiffness term for the i-th element
380  Values computeStiffnessTerm(int i) override{
381  Eigen::Matrix<double, 1, PHDIM> grad = this->grad_w.row(i);
382  Eigen::Matrix<double, 1, PHDIM> localLagrangian = lagrangians.row(i);
383  double q_norm = this->anisotropicNorm(grad - localLagrangian/this->r);
384  double stiffnessCoeff = (1.0 - q_norm) / ((1.0 + this->r)*q_norm + this->gamma);
385  return stiffnessCoeff * (this->mesh.getLocalStiffnessMatrices()[i] * this->local_w);
386  }
387 
388  // Compute the reaction term for the i-th element
389  Values computeReactionTerm(int i) override {
390  Eigen::Matrix<double, 1, PHDIM> grad = this->grad_w.row(i);
391  Eigen::Matrix<double, 1, PHDIM> localLagrangian = lagrangians.row(i);
392  double q_norm = this->anisotropicNorm(grad - localLagrangian/r);
393  double reactionCoeff = (q_norm - 1.0) / ((r * (1.0+r) * q_norm) + this->gamma);
394 
395  Values uu(PHDIM+1);
396  for (int j = 0; j < PHDIM+1; j++) {
397  for (int k = 0; k < PHDIM+1; k++) {
398  uu(j) += this->mesh.getLocalReactionMatrices()[i][j][k].dot(localLagrangian.transpose()); // (PHDIM x 1) x (PHDIM x 1)
399  }
400  }
401  return reactionCoeff * uu;
402  }
403 
404  // Update the Lagrangian multipliers for each element
405  void updateLagrangians(const Values& z) override {
406  Eigen::Matrix<double, Eigen::Dynamic, 1> q_norm(this->grad_w.rows());
407 
408  // Compute the norm of the difference between the gradient and the Lagrangian
409  for (int i = 0; i < this->grad_w.rows(); ++i) {
410  Eigen::Matrix<double, 1, PHDIM> diff = this->grad_w.row(i) - lagrangians.row(i) / this->r;
411  q_norm(i) = this->anisotropicNorm(diff);
412  }
413 
414  // Compute coefficients for Lagrangian and gradient updates
415  Eigen::Matrix<double, Eigen::Dynamic, 1> c = (1.0 + this->r * q_norm.array()) / ((1 + this->r) * q_norm.array());
416  Eigen::Matrix<double, Eigen::Dynamic, 1> lagrangianCoeff = 1 - c.array();
417  Eigen::Matrix<double, Eigen::Dynamic, 1> gradzCoeff = this->r * c;
418 
419  // Local variables for each element
420  Values local_z = Values::Zero(this->mesh.getConnectivity().rows());
421  Eigen::Matrix<double, Eigen::Dynamic, PHDIM> grad_z(this->mesh.getNumElements(), PHDIM);
422 
423  // Update the Lagrangian multipliers for each element
424  for(int i = 0; i < this->mesh.getGradientCoeff().size(); i++) {
425  // Collect local z values
426  for (int j = 0; j < PHDIM+1; j++) {
427  local_z(j) = z(this->mesh.getConnectivity().col(i)(j));
428  }
429 
430  // Compute the gradient of z
431  grad_z.row(i) = apsc::computeGradient<PHDIM>(this->mesh.getGradientCoeff()[i], local_z);
432 
433  // Update the Lagrangian multipliers
434  lagrangians.row(i) = lagrangianCoeff(i) * lagrangians.row(i) + gradzCoeff(i) * grad_z.row(i);
435  }
436  }
