421  void updateBoundaryNodes(const int& boundaryCondition) {
422  std::map<std::set<int>, int> faceCount;
423  for (int i = 0; i < numElements; ++i) {
424  // Each element has 4 faces
425  for (int j = 0; j < 4; ++j) {
426  std::set<int> face;
427  for (int k = 0; k < 4; ++k) {
428  if (k != j) {
429  face.insert(connectivity(k, i)); // Assuming connectivity is a 2D array or similar structure
430  }
431  }
432  faceCount[face]++;
433  }
434  }
435 
436  std::set<int> boundaryNodesSet;
437  for (const auto& entry : faceCount) {
438  if (entry.second == 1) { // Boundary face
439  for (int node : entry.first) {
440  boundaryNodesSet.insert(node);
441  }
442  }
443  }
444 
445  boundaryNodes.clear();
446  switch (boundaryCondition) {
447  case 1:
448  // Include only nodes with z=0
449  for (int node : boundaryNodesSet) {
450  if (nodes(2, node) == 0) { // Assuming node position z is at index 2 in nodes
451  boundaryNodes.push_back(node);
452  }
453  }
454  break;
455  case 2:
456  // Include only the center node of the face with z=0
457  {
458  std::vector<int> z0Nodes;
459  for (int node : boundaryNodesSet) {
460  if (nodes(2, node) == 0) { // Check for z=0 nodes
461  z0Nodes.push_back(node);
462  }
463  }
464 
465  if (!z0Nodes.empty()) {
466  // Calculate the geometric center by averaging coordinates
467  double centerX = 0.0, centerY = 0.0;
468  for (int node : z0Nodes) {
469  centerX += nodes(0, node); // X coordinate
470  centerY += nodes(1, node); // Y coordinate
471  }
472  centerX /= z0Nodes.size();
473  centerY /= z0Nodes.size();
474 
475  // Find the node closest to the calculated center
476  int centerNode = z0Nodes[0];
477  double minDistance = std::numeric_limits<double>::max();
478  for (int node : z0Nodes) {
479  double distance = std::sqrt(std::pow(nodes(0, node) - centerX, 2) + std::pow(nodes(1, node) - centerY, 2));
480  if (distance < minDistance) {
481  minDistance = distance;
482  centerNode = node;
483  }
484  }
485  boundaryNodes.push_back(centerNode);
486  }
487  }
488  break;
489  case 3:
490  // Include only the vertex at the origin (0, 0, 0)
491  for (int node : boundaryNodesSet) {
492  if (nodes(0, node) == 0 && nodes(1, node) == 0 && nodes(2, node) == 0) {
493  boundaryNodes.push_back(node);
494  break; // No need to search further, there is only one origin
495  }
496  }
497  break;
498  case 4:
499  // Include all boundary nodes
500  boundaryNodes.assign(boundaryNodesSet.begin(), boundaryNodesSet.end());
501  break;
502  default:
503  std::cerr << "Invalid option. Please run the program again with a valid choice.\n";
504  return; // Exit with an error code
505  }
506  }
